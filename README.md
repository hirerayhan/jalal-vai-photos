    function startCountdown() {
            // Set dynamic future date: 3 days from current time to ensure it always works
            const now = new Date();
            const targetDate = new Date(now.getTime() + (3 * 24 * 60 * 60 * 1000) + (11 * 60 * 60 * 1000) + (30 * 60 * 1000)); 

            function updateTimer() {
                const currentTime = new Date();
                const difference = targetDate - currentTime;

                if (difference <= 0) {
                    document.getElementById('days').innerText = "00";
                    document.getElementById('hours').innerText = "00";
                    document.getElementById('minutes').innerText = "00";
                    document.getElementById('seconds').innerText = "00";
                    return;
                }

                const days = Math.floor(difference / (1000 * 60 * 60 * 24));
                const hours = Math.floor((difference % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
                const minutes = Math.floor((difference % (1000 * 60 * 60)) / (1000 * 60));
                const seconds = Math.floor((difference % (1000 * 60)) / 1000);

                // Add leading zeros and translate to Bengali Numerals
                const bnDays = String(days).padStart(2, '0').replace(/\d/g, d => '০১২৩৪৫৬৭৮৯'[d]);
                const bnHours = String(hours).padStart(2, '0').replace(/\d/g, d => '০১২৩৪৫৬৭৮৯'[d]);
                const bnMinutes = String(minutes).padStart(2, '0').replace(/\d/g, d => '০১২৩৪৫৬৭৮৯'[d]);
                const bnSeconds = String(seconds).padStart(2, '0').replace(/\d/g, d => '০১২৩৪৫৬৭৮৯'[d]);

                document.getElementById('days').innerText = bnDays;
                document.getElementById('hours').innerText = bnHours;
                document.getElementById('minutes').innerText = bnMinutes;
                document.getElementById('seconds').innerText = bnSeconds;
            }

            // Initial call and interval
            updateTimer();
            setInterval(updateTimer, 1000);
        }

        /**
         * 5. PARTICLES.JS ENGINE (Custom built mini version to avoid external dependencies)
         * Creates a lightweight node/vertex connection effect in the background
         */
        class ParticleNetwork {
            constructor(canvas) {
                this.canvas = canvas;
                this.ctx = this.canvas.getContext('2d');
                this.particles = [];
                this.resize();
                
                // Options
                this.options = {
                    particleColor: 'rgba(57, 255, 20, 0.4)',
                    lineColor: 'rgba(57, 255, 20, 0.15)',
                    particleAmount: Math.floor((window.innerWidth * window.innerHeight) / 15000),
                    defaultRadius: 1.5,
                    variantRadius: 2,
                    linkRadius: 150,
                    defaultSpeed: 0.3,
                    variantSpeed: 0.5
                };

                this.init();
                window.addEventListener('resize', () => this.resize());
                this.animate();
            }

            resize() {
                this.canvas.width = window.innerWidth;
                this.canvas.height = window.innerHeight;
            }

            init() {
                this.particles = [];
                for (let i = 0; i < this.options.particleAmount; i++) {
                    this.particles.push(new Particle(this));
                }
            }

            animate() {
                requestAnimationFrame(() => this.animate());
                this.ctx.clearRect(0, 0, this.canvas.width, this.canvas.height);
                
                for (let i = 0; i < this.particles.length; i++) {
                    this.particles[i].update();
                    this.particles[i].draw();
                }

                this.drawLines();
            }

            drawLines() {
                for (let i = 0; i < this.particles.length; i++) {
                    for (let j = i + 1; j < this.particles.length; j++) {
                        let distance = this.checkDistance(this.particles[i], this.particles[j]);
                        if (distance < this.options.linkRadius) {
                            let opacity = 1 - (distance / this.options.linkRadius);
                            this.ctx.beginPath();
                            this.ctx.strokeStyle = `rgba(57, 255, 20, ${opacity * 0.2})`;
                            this.ctx.lineWidth = 1;
                            this.ctx.moveTo(this.particles[i].x, this.particles[i].y);
                            this.ctx.lineTo(this.particles[j].x, this.particles[j].y);
                            this.ctx.stroke();
                        }
                    }
                }
            }

            checkDistance(p1, p2) {
                return Math.sqrt(Math.pow(p2.x - p1.x, 2) + Math.pow(p2.y - p1.y, 2));
            }
        }

        class Particle {
            constructor(network) {
                this.network = network;
                this.x = Math.random() * this.network.canvas.width;
                this.y = Math.random() * this.network.canvas.height;
                this.speed = this.network.options.defaultSpeed + Math.random() * this.network.options.variantSpeed;
                this.directionAngle = Math.floor(Math.random() * 360);
                this.vector = {
                    x: Math.cos(this.directionAngle) * this.speed,
                    y: Math.sin(this.directionAngle) * this.speed
                };
                this.radius = this.network.options.defaultRadius + Math.random() * this.network.options.variantRadius;
            }

            update() {
                this.border();
                this.x += this.vector.x;
                this.y += this.vector.y;
            }

            border() {
                if (this.x >= this.network.canvas.width || this.x <= 0) {
                    this.vector.x *= -1;
                }
                if (this.y >= this.network.canvas.height || this.y <= 0) {
                    this.vector.y *= -1;
                }
                if (this.x > this.network.canvas.width) this.x = this.network.canvas.width;
                if (this.y > this.network.canvas.height) this.y = this.network.canvas.height;
                if (this.x < 0) this.x = 0;
                if (this.y < 0) this.y = 0;
            }

            draw() {
                this.network.ctx.beginPath();
                this.network.ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2);
                this.network.ctx.closePath();
                this.network.ctx.fillStyle = this.network.options.particleColor;
                this.network.ctx.fill();
            }
        }
